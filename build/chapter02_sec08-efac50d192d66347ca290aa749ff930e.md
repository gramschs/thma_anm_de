````{admonition} Lösung
```python
import numpy as np
import matplotlib.pyplot as plt

# Eingabe
stunden    = np.arange(0, 24)
windgeschw = np.array([3.2, 2.8, 2.5, 2.1, 2.4, 3.0, 4.1, 5.3,
                        6.2, 7.0, 7.8, 8.1, 7.5, 6.9, 6.3, 5.8,
                        5.1, 4.7, 4.2, 3.8, 3.5, 3.3, 3.1, 2.9])

# Verarbeitung
idx_max   = np.argmax(windgeschw)
mittelwert = np.mean(windgeschw)

# Ausgabe
fig, ax = plt.subplots(figsize=(10, 5))

ax.plot(stunden, windgeschw, color='steelblue', linewidth=1.5,
        marker='o', markersize=4, label='Windgeschwindigkeit')
ax.scatter(stunden[idx_max], windgeschw[idx_max],
           marker='*', s=200, color='red', zorder=5,
           label=f'Maximum: {windgeschw[idx_max]:.1f} m/s um {stunden[idx_max]}:00 Uhr')
ax.axhline(y=mittelwert, linestyle='dashed', color='gray',
           label=f'Mittelwert: {mittelwert:.1f} m/s')

ax.set_xlabel('Uhrzeit in h')
ax.set_ylabel('Windgeschwindigkeit in m/s')
ax.set_title('Windgeschwindigkeit im Tagesverlauf')
ax.legend()
ax.grid(True)

plt.tight_layout()
plt.show()
```

Das Maximum tritt um 11:00 Uhr mit 8.1 m/s auf. Der typische Tagesgang
des Windes zeigt niedrige Geschwindigkeiten in den frühen Morgenstunden
und einen Anstieg bis zum Mittag, was durch die solare Erwärmung und die
damit verbundene Konvektion erklärt wird.
````

````{admonition} Lösung
```python
import numpy as np
import matplotlib.pyplot as plt

# Eingabe
proben    = np.array([1, 2, 3, 4, 5])
bruchspg  = np.array([42.3, 38.7, 51.2, 45.8, 39.1])
std_bruch = np.array([ 2.1,  3.4,  1.8,  2.7,  3.0])

np.random.seed(3)
alle_werte = np.concatenate([
    np.random.normal(m, s, 10) for m, s in zip(bruchspg, std_bruch)
])

# Verarbeitung
mittelwert = np.mean(bruchspg)

# Ausgabe
fig, ax = plt.subplots(nrows=1, ncols=2, figsize=(12, 5))

ax[0].errorbar(proben, bruchspg, yerr=std_bruch,
               fmt='s', color='steelblue', capsize=6,
               linewidth=1.5, label='Bruchspannung ± Std')
ax[0].axhline(y=mittelwert, linestyle='dashed', color='gray',
              label=f'Mittelwert: {mittelwert:.1f} MPa')
ax[0].set_ylim(30, 60)
ax[0].set_xlabel('Probennummer')
ax[0].set_ylabel('Bruchspannung in MPa')
ax[0].set_title('Bruchspannungen der Kunststoffproben')
ax[0].legend()
ax[0].grid(True)

ax[1].hist(alle_werte, bins=15, color='steelblue', edgecolor='white')
ax[1].set_xlabel('Bruchspannung in MPa')
ax[1].set_ylabel('Haeufigkeit')
ax[1].set_title('Verteilung aller Messwerte (N=50)')
ax[1].grid(True, axis='y')

plt.tight_layout()
plt.show()
```

Probe 3 zeigt mit 51.2 MPa die höchste Bruchspannung und gleichzeitig die
kleinste Streuung, was auf ein hochwertiges und gleichmäßiges Material
hindeutet. Das Histogramm zeigt eine breite Verteilung der Einzelwerte, weil
es Messungen aus fünf Proben mit unterschiedlichen Mittelwerten enthält.
````

````{admonition} Lösung

```python
import numpy as np
import matplotlib.pyplot as plt

# Eingabe
regen   = np.array([ 82. ,  92.5,  83.2,  97.7, 131.9, 141.3,
                     165.4, 140. , 126.7, 114.2,  98.3,  86.1])
schirme = np.array([15, 25, 17, 28, 41, 47, 50, 46, 37, 33, 26, 18])
monate  = ['Jan', 'Feb', 'Mar', 'Apr', 'Mai', 'Jun',
           'Jul', 'Aug', 'Sep', 'Okt', 'Nov', 'Dez']

# Verarbeitung: Lineare Regression
N   = len(regen)
x   = regen.astype(float)
y   = schirme.astype(float)
m   = (N * np.sum(x * y) - np.sum(x) * np.sum(y)) / \
      (N * np.sum(x**2) - np.sum(x)**2)
b   = (np.sum(x**2) * np.sum(y) - np.sum(x) * np.sum(x * y)) / \
      (N * np.sum(x**2) - np.sum(x)**2)

x_linie     = np.linspace(75, 175, 200)
y_linie     = m * x_linie + b
vorhersage  = m * 162 + b

print(f'Steigung:      {m:.3f} Schirme/mm')
print(f'Achsenabschnitt: {b:.2f} Schirme')
print(f'Vorhersage fuer 162 mm: {vorhersage:.1f} Schirme')

# Ausgabe
fig, ax = plt.subplots(nrows=2, ncols=2, figsize=(14, 10))

# Oben links: Scatter + Regression
scatter = ax[0, 0].scatter(regen, schirme,
                            c=np.arange(12), cmap='hsv',
                            s=80, zorder=5, label='Monatsdaten')
ax[0, 0].plot(x_linie, y_linie, color='steelblue', linewidth=1.5,
              label=f'Regression: y = {m:.3f}x + {b:.2f}')
ax[0, 0].scatter(162, vorhersage, marker='*', s=250, color='red',
                 zorder=6, label=f'Vorhersage 162 mm: {vorhersage:.1f}')
fig.colorbar(scatter, ax=ax[0, 0],
             ticks=np.arange(12),
             label='Monat').set_ticklabels(monate)
ax[0, 0].set_xlabel('Regenfall in mm')
ax[0, 0].set_ylabel('Schirmverkäufe')
ax[0, 0].set_title('Regression: Regen vs. Schirmverkauf')
ax[0, 0].legend(fontsize=8)
ax[0, 0].grid(True)

# Unten links: leer (oder Legende)
ax[1, 0].axis('off')

# Oben rechts: monatlicher Regenfall
ax[0, 1].plot(np.arange(12), regen, color='steelblue',
              marker='o', markersize=5, linewidth=1.5)
ax[0, 1].set_ylabel('Regenfall in mm')
ax[0, 1].set_title('Monatlicher Regenfall')
ax[0, 1].set_xticks(np.arange(12))
ax[0, 1].set_xticklabels(monate, rotation=45)
ax[0, 1].grid(True)

# Unten rechts: monatliche Schirmverkäufe
ax[1, 1].plot(np.arange(12), schirme, color='tomato',
              marker='o', markersize=5, linewidth=1.5)
ax[1, 1].set_ylabel('Schirmverkäufe')
ax[1, 1].set_title('Monatliche Schirmverkäufe')
ax[1, 1].set_xticks(np.arange(12))
ax[1, 1].set_xticklabels(monate, rotation=45)
ax[1, 1].grid(True)

plt.tight_layout()
plt.show()
```

Ausgabe:
```
Steigung:        0.440 Schirme/mm
Achsenabschnitt: -16.17 Schirme
Vorhersage fuer 162 mm: 55.1 Schirme
```

**Teil 4:** Die lineare Regression beschreibt den Zusammenhang gut. Der
Korrelationstrend ist klar erkennbar: Mehr Regen geht mit mehr
Schirmverkäufen einher. Die Sommermonate Juli und August zeigen die
höchsten Werte in beiden Größen. Die Vorhersage von ca. 55 Schirmen
bei 162 mm Regen erscheint plausibel, liegt aber leicht über dem
bisherigen Maximum von 50. Eine lineare Regression setzt voraus, dass
der Zusammenhang tatsächlich linear ist und außerhalb des beobachteten
Bereichs gültig bleibt, was nicht immer der Fall ist.
````
